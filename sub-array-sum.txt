#include<stdio.h>
int subarray(int arr[],int size)
{
   int i,j,sum;
   for(i=0;i<size;i++)
   {
       sum=0;
       for(j=i;j<size;j++)
       {
           sum=sum+arr[j];
           if(sum==0)
           {
               return 1;
           }
       }
   }
   return 0;
}
int main()
{
    int arr[]={4,2,-3,1,6};
    int size=sizeof(arr)/sizeof(arr[0]);
    if(subarray(arr,size))
    {
        printf("true\n");
    }
    else
    {
        printf("false");
    }
}
